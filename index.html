// ESP32 firmware driver polling sensor registers via SPI
#include <SPI.h>
#define REG_SENSOR_DATA 0x42
#define PIN_CS 15

void setup() {
  SPI.begin();
  pinMode(PIN_CS, OUTPUT);
  digitalWrite(PIN_CS, HIGH);
  Serial.begin(115200);
}

uint16_t readSensor() {
  digitalWrite(PIN_CS, LOW);
  SPI.transfer(REG_SENSOR_DATA);
  uint8_t high = SPI.transfer(0x00);
  uint8_t low = SPI.transfer(0x00);
  digitalWrite(PIN_CS, HIGH);
  return (high << 8) | low;
}

void loop() {
  uint16_t val = readSensor();
  Serial.printf("Sensor Value: %d\n", val);
  delay(100);
}
